// For every image, we check if the degree of the corresponding point 
// on X_1(ell^n) is greater than the maximum of #Delta/2 times the 
// genus of X_Delta(ell^n), as Delta ranges over all subgroups 
// corresponding to intermediate modular curves of positive genus. 
// This maximum value is 8, 3, and 10, for ell^n equal to 27, 25, 
// and 32, respectively. We collect the pairs which do not satisfy 
// this condition in the set Images.

RemainingImages := {* *};

for Pair in ImageList do
Gens := Pair[2];
if Index(ImageList, Pair) le 9 then
elln := 25;
end if;
if Index(ImageList, Pair) ge 10 and Index(ImageList, Pair) le 12 then
elln := 27;
end if;
if Index(ImageList, Pair) ge 13 then
elln := 32;
end if;
G := sub<GL(2,Integers(elln))|Gens>;
Gt := sub<GL(2,BaseRing(G)) | [Transpose(g):g in Generators(G)]>;
m := Modulus(BaseRing(Gt));
H := sub<GL(2,Integers(m))|Gt,-Gt!1>;
orb := Orbits(H);
orbm := [ m ne 2 select #x div 2 else #x : x in orb | #Parent(x[1][1]) div GCD([#Parent(x[1][1]), Integers()!x[1][2], Integers()!x[1][1]]) eq m];
degrees := SequenceToMultiset(orbm);
if Index(ImageList, Pair) le 9 and Min(degrees) le 8 then
Include(~RemainingImages, Pair);
end if;
if Index(ImageList, Pair) ge 10 and Index(ImageList, Pair) le 12 and Min(degrees) le 3 then
Include(~RemainingImages, Pair);
end if;
if Index(ImageList, Pair) ge 13 and Min(degrees) le 10 then
Include(~RemainingImages, Pair);
end if;
end for;

// For each pair in RemainingImages, we compute the genus of the 
// modular curve corresponding to the mod-ell image.

PositiveGenusImages := {* *};

for Pair in RemainingImages do
Gens := Pair[2];
if Index(ImageList, Pair) le 9 then
elln := 25;
end if;
if Index(ImageList, Pair) ge 10 and Index(ImageList, Pair) le 12 then
elln := 27;
end if;
if Index(ImageList, Pair) ge 13 then
elln := 32;
end if;
ell := Factorization(elln)[1][1];
G := sub<GL(2,Integers(elln))|Gens>;
Gmod := ChangeRing(G,Integers(ell));
g, L := GL2Genus(Gmod);
if g ge 1 then
Include(~PositiveGenusImages, Pair);
end if;
end for;

// For all images in RemainingImages, the genus of the modular curve 
// corresponding to the mod-ell image is zero.

assert #PositiveGenusImages eq 0;
